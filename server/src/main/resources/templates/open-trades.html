function openDetailModal(accountId) { const dataDiv = document.getElementById('trades-data-' + accountId); if (!dataDiv) return; const modalBody = document.getElementById('modal-body-content'); modalBody.innerHTML = dataDiv.innerHTML; const modal = document.getElementById('detail-modal'); modal.style.display = 'flex'; // Ensure display flex before opacity transition // Small delay to allow display change to render before adding active class for transition requestAnimationFrame(() => { modal.classList.add('active'); }); document.body.style.overflow = 'hidden'; // Prevent background scrolling } function closeDetailModal(event) { if (event && event.target !== event.currentTarget && event.target.className !== 'modal-close') { return; // Clicked inside modal content (if called via overlay click, this check is redundant due to stopPropagation, but safe) } const modal = document.getElementById('detail-modal'); modal.classList.remove('active'); document.body.style.overflow = ''; // Restore scrolling // Wait for transition to finish before hiding setTimeout(() => { modal.style.display = 'none'; document.getElementById('modal-body-content').innerHTML = ''; }, 300); }
